#histogram for gdp data 
gdp <- ggplot(gapminder, aes(x = gdpPercap)) +
  geom_histogram(fill = "lightblue", color = "black", binwidth = 2000) +
  labs(title = "Histogram of GDP per Capita", x = "GDP per Capita", y = "Count")
